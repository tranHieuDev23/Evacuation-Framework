    \begin{enumerate}
        \item label
        \item Next Node
        \item nextOption
        \item Previous Node
        \item AdjacentEdges
        \item Number of t(s) coming people
        \item weight
    \end{enumerate}
    \textbf{Edge}
    \begin{enumerate}
        \item label
        \item Node1, Node2
        \item Length
        \item Width
        \item Trustness
        \item NumberOfPeople
        \item Capacity
        \item weight
    \end{enumerate}
    \begin{algorithmic}
    \Procedure{My Procedure}{}
        \State Cho các Edge nối từ exitNode với root vào heap
        \While{$heap.size > 0$}
        \State u = heap.pop()
        \State u.label = true
        \For{v in u.adjacence and v.label = false} 
        \State s = v.getFutureNode(u)
        \State s.numberOfComingPeople += v.people
        \State calculateWeight(v, u, path(u, s), path(s, root))
        \EndFor
        \EndWhile
    \EndProcedure
    \end{algorithmic}
